  SELECT company_id FROM profiles WHERE id = auth.uid() AND active = true;

-- ----------------------------------------------------------------------------
-- 3. Recarregar schema cache do PostgREST
-- ----------------------------------------------------------------------------
NOTIFY pgrst, 'reload schema';
