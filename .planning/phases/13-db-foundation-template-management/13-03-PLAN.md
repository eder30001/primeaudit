Future<void> _load() async {
  setState(() { _isLoading = true; _error = null; });
  try {
    final results = await Future.wait([
      _service.getByCategory('industrial'),
      _service.getByCategory('transportadora'),
      _service.getOwned(),
    ]);
    if (mounted) setState(() {
      _industrial = results[0];
      _transportadora = results[1];
      _owned = results[2];
    });
  } catch (e) {
    if (mounted) setState(() => _error = 'Erro ao carregar templates. Puxe para atualizar.');
  } finally {
    if (mounted) setState(() => _isLoading = false);
  }
}

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-01 | Tampering | _confirmDelete UI guard | mitigate | Seed cards never show PopupMenuButton (template.isSeed check). Even if UI is bypassed, RLS DELETE policy blocks at DB. |
| T-13-02 | Information Disclosure | _ChecklistTemplateCard visibility | mitigate | getByCategory and getOwned queries filtered server-side by RLS; client only renders what the server returns. |
| T-13-03 | Tampering | _openEdit UI guard | mitigate | Edit option only shown when createdBy == currentUserId. RLS UPDATE policy is the authoritative guard. |
| T-13-09 | Spoofing | currentUserId from auth.currentUser | accept | Supabase auth session managed by SDK; currentUser?.id is the verified JWT sub claim. Low risk. |
