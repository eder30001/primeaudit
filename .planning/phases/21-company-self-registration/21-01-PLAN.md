```sql
-- =============================================================================
-- Migration: adiciona colunas de billing em companies + RPC de self-registration
-- Data: 2026-05-15
-- Idempotente: ADD COLUMN IF NOT EXISTS, DROP/ADD CONSTRAINT, CREATE OR REPLACE
-- Requerido por: Phase 21 (self-registration) + Phase 22 (billing Asaas)
-- =============================================================================

-- ----------------------------------------------------------------------------
-- 1. companies.status — estado do ciclo de vida da empresa
--    DEFAULT 'active' preserva empresas existentes sem trial
-- ----------------------------------------------------------------------------
ALTER TABLE companies ADD COLUMN IF NOT EXISTS status TEXT NOT NULL DEFAULT 'active';

ALTER TABLE companies DROP CONSTRAINT IF EXISTS companies_status_check;
ALTER TABLE companies
  ADD CONSTRAINT companies_status_check
  CHECK (status IN ('trial', 'active', 'payment_pending', 'suspended'));

-- ----------------------------------------------------------------------------
-- 2. companies.trial_expires_at — nullable: só definido em empresas auto-criadas
-- ----------------------------------------------------------------------------
ALTER TABLE companies ADD COLUMN IF NOT EXISTS trial_expires_at TIMESTAMPTZ;

-- ----------------------------------------------------------------------------
-- 3. companies.license_expires_at — nullable: definido pelo webhook Asaas (Phase 22)
-- ----------------------------------------------------------------------------
ALTER TABLE companies ADD COLUMN IF NOT EXISTS license_expires_at TIMESTAMPTZ;

-- ----------------------------------------------------------------------------
-- 4. Patch no trigger handle_new_user
--    Impede escalada de privilégio: role via raw_user_meta_data é clamped a
--    'auditor' ou 'adm'. 'superuser' e 'dev' só podem ser criados por superuser
--    via painel admin — nunca via signUp público.
-- ----------------------------------------------------------------------------
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (id, full_name, email, role, company_id)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'full_name', ''),
    NEW.email,
    COALESCE(
      CASE
        WHEN NEW.raw_user_meta_data->>'role' IN ('auditor', 'adm')
        THEN NEW.raw_user_meta_data->>'role'
        ELSE 'auditor'
      END,
      'auditor'
    ),
    (NEW.raw_user_meta_data->>'company_id')::UUID
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ----------------------------------------------------------------------------
-- 5. Função SECURITY DEFINER para criação de empresa no cadastro público
--    Chamada via rpc() com chave anon — contorna RLS de companies.
--    Retorna o UUID da empresa criada para o Flutter usar no signUp seguinte.
--
--    Guards (servidor — não dependem do cliente):
--      - CNPJ normalizado para 14 dígitos
--      - CNPJ deve ter exatamente 14 dígitos
--      - CNPJ não pode já existir (UNIQUE constraint + guard explícito)
--      - Nome não pode ser vazio
-- ----------------------------------------------------------------------------
CREATE OR REPLACE FUNCTION create_company_for_registration(
  p_cnpj    TEXT,
  p_name    TEXT,
  p_segment TEXT DEFAULT 'industrial',
  p_modules TEXT[] DEFAULT ARRAY['auditoria', 'checklist']
)
RETURNS UUID
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_clean_cnpj TEXT;
  v_company_id  UUID;
BEGIN
  -- Normalizar CNPJ (remove pontos, barras, hífens, espaços)
  v_clean_cnpj := regexp_replace(p_cnpj, '[.\-/\s]', '', 'g');

  -- Guard: CNPJ deve ter exatamente 14 dígitos após normalização
  IF length(v_clean_cnpj) != 14 THEN
    RAISE EXCEPTION 'CNPJ deve ter 14 dígitos';
  END IF;

  -- Guard: CNPJ já cadastrado (evita empresa órfã em race condition)
  IF EXISTS (SELECT 1 FROM companies WHERE cnpj = v_clean_cnpj) THEN
    RAISE EXCEPTION 'CNPJ já cadastrado no sistema';
  END IF;

  -- Guard: nome obrigatório
  IF trim(p_name) = '' THEN
    RAISE EXCEPTION 'Nome da empresa é obrigatório';
  END IF;

  -- Criar empresa em status trial com 30 dias de avaliação
  INSERT INTO companies (cnpj, name, segment, modules, status, trial_expires_at, active)
  VALUES (
    v_clean_cnpj,
    trim(p_name),
    p_segment,
    p_modules,
    'trial',
    now() + interval '30 days',
    true
  )
  RETURNING id INTO v_company_id;

  RETURN v_company_id;
END;
$$;

-- Permite que clientes não-autenticados (chave anon) chamem a função via rpc()
-- Padrão Supabase para operações públicas que requerem bypass de RLS controlado
GRANT EXECUTE ON FUNCTION create_company_for_registration(TEXT, TEXT, TEXT, TEXT[]) TO anon;

NOTIFY pgrst, 'reload schema';
```
